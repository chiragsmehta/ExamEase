Table  user{
  id int [primary key]
  username string [not null]
  email string [not null]
  password string [not null]
  user_role enum
}

Table exam{
  id int [primary key]
  name string [not null]
  start_date_time datetime [not null]
  end_date_time datetime [not null]
  duration int [not null]
  total_marks int
}


Table programming_question{
  id int [primary key]
  description string
  question string
  exam_id int
  difficulty_level_id int
  answer string
  marks int
}

Table difficulty_level{
  id int [primary key]
  name string
}


Table mcq_question{
  id int [primary key]
  question string
  exam_id int
  difficulty_level_id int
  marks int
}

Table option{
  id int [primary key]
  value string
  is_correct boolean
  mcq_question_id int
}

Table exam_enrollment{
  id int [primary key]
  user_id int
  exam_id int
}

Table user_exam_status{
  id int [primary key]
  exam_enrollment_id int
  start_date_time int
  submission_date_time int
  has_submitted boolean
}

Table user_mcq_question_submission{
  id int [primary key]
  user_id int
  mcq_question_id int
  option_id int
}

Table user_programming_question_submission{
  id int [primary key]
  user_id int
  programming_question_id int
  user_solution string
}

Table image{
  id int [primary key]
  url string
  alt_text string
}

Table mcq_question_image{
  id int [primary key]
  image_id int
  mcq_question_id int
}

Table programming_question_image{

}

  option_id int
